1 /**
  2 * @namespace A utility for logging messages to the console
  3 * @name logger
  4 */
  5 define([],function() {
  6 	var logger = {};
  7 	
  8 	var loglevels = { error: 0, warn: 1, info: 2, debug: 3 };
  9 	var loglevel = loglevels.debug;
 10 	
 11 	/**
 12 	* Set the loggers log level
 13 	* @memberOf logger
 14 	* @param level {String}: One of [error,warn,info,debug]
 15 	*/
 16 	logger.setLogLevel = function(level) {
 17 		loglevel = loglevels.level;
 18 	}
 19 	
 20 	/**
 21 	* Log a debug message (white)
 22 	* @memberOf logger
 23 	* @param str {String}: The string to write
 24 	*/
 25 	logger.debug = function(str) {
 26 		if(loglevel >= loglevels.debug) {
 27 			console.debug(str);
 28 		}
 29 	}
 30 	
 31 	/**
 32 	* Log an info message (blue)
 33 	* @memberOf logger
 34 	* @param str {String}: The string to write
 35 	*/
 36 	logger.info = function(str) {
 37 		if(loglevel >= loglevels.info) {
 38 			console.info(str);
 39 		}
 40 	}
 41 	
 42 	/**
 43 	* Log a warning message (yellow)
 44 	* @memberOf logger
 45 	* @param str {String}: The string to write
 46 	*/
 47 	logger.warn = function(str) {
 48 		if(loglevel >= loglevels.warn) {
 49 			console.warn(str);
 50 		}
 51 	}
 52 	
 53 	/**
 54 	* Log an error message (red)
 55 	* @memberOf logger
 56 	* @param str {String}: The string to write
 57 	*/
 58 	logger.error = function(str) {
 59 		if(loglevel >= loglevels.err) {
 60 			console.error(str);
 61 		}
 62 	}
 63 	
 64 	var test = function() {
 65 		logger.debug("debugging info");
 66 		logger.info("info");
 67 		logger.warn("warning");
 68 		//logger.error("error");
 69 		logger.info("initialized logger");
 70 	}
 71 	
 72 	test();
 73 	
 74 	return logger;
 75 });
