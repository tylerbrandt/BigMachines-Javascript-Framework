  8 	test_util.generateRandomColor = function() {
  9 		var max = 255;
 10 		var red = Math.floor(Math.random() * max);
 11 		var green = Math.floor(Math.random() * max);
 12 		var blue = Math.floor(Math.random() * max);
 13 		return "rgb(" + red + "," + green + "," + blue + ")";
 14 	}
 15 
 16 	/** 
 17 	 * Add a border of the specified {color} to {elem}
 18 	 * @param elem The element to apply the border to
 19 	 * @param color The color of the border
 20 	 */
 21 	test_util.apply_border = function(elem,color) {
 22 		jQuery(elem).css("border","2px solid " + color);
 23 	};
 24 
 25 	/** 
 26 	 * Generates a qunit test that verifies that at least one of {elem} exists and applies a border
 27 	 * @param elem The element to apply the border to

 30 	 */
 31 	test_util.border_test = function(elem,testName,color) {
 32 		if(typeof color === "undefined") {

 34 		}

 36 		test("<span style='border: 1px solid " + color + "' id='" + testID + "'>" + testName + "</span>",function() {

 40 		});
