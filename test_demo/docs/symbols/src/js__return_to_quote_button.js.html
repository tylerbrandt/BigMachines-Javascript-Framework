1 /**
  2 * @namespace A module to add a "Return To Quote" button on the homepage, based on a recent quote.
  3 *
  4 * @name return_to_quote_button
  5 *
  6 * @requires manager
  7 * @requires jquery.cookie
  8 **/
  9 define(["manager","logger"], function(mgr,logger) {
 10 	mgr.register("return_to_quote_button");
 11 	var return_to_quote_button = {};
 12 	
 13 	require(["jquery.cookie"],function() {
 14 		/**
 15 		* Called from homepage. Shows a return to quote button on the page, contingent
 16 		* on a cookie created by set_cookie_in_commerce.
 17 		* 
 18 		* Works out of the box with Quickstart Version 11. With older versions, you have to include a callback
 19 		* to build the element for you.
 20 		*
 21 		* @memberOf return_to_quote_button
 22 		* @param callback(urlToQuote) {Function} Optional. If included, then this callback will be called instead of showing the default return to quote element
 23 		*/
 24 		return_to_quote_button.add_button_to_homepage = function(callback,context) {
 25 			logger.debug("adding button to homepage");
 26 			context = context || document;
 27 			var last_trans_cookie = jQuery.cookie("last_transaction");
 28 			//if cookie is set add functionality to link
 29 			if( last_trans_cookie !== null && last_trans_cookie !== "" && last_trans_cookie !== "-1" ) {
 30 				var url = '/commerce/buyside/document.jsp?formaction=performAction&';
 31 				url += last_trans_cookie;
 32 
 33 				//version 11 default behavior
 34 				if(typeof callback !== "function") {
 35 					logger.debug("using v11 behavior");
 36 					jQuery(".return-to-quote",context).attr("href", url);
 37 					jQuery(".return-to-quote",context).show();
 38 					
 39 					//parts search doesn't exist (this class is set condtionally in xsl)
 40 					if(jQuery(".return-to-quote-wrapper",context).length){
 41 						jQuery(".return-to-quote-wrapper",context).show();
 42 					}
 43 					else{ //parts search exists
 44 						jQuery(".return-to-quote-pipe",context).show();
 45 					}
 46 				} else {
 47 					callback(url,context);
 48 				}
 49 			}
 50 		};
 51 
 52 
 53 		/**
 54 		* Called from commerce. Sets a cookie used to Return to Quote from homepage.
 55 		* @memberOf return_to_quote_button
 56 		* @requires commerce_ids
 57 		*/
 58 		return_to_quote_button.set_cookie_in_commerce = function() {
 59 			require(["commerce_ids"],function(c_ids) {
 60 				//get ids off page
 61 				var ids = c_ids.get_ids();
 62 				//builds url
 63 				var last_transaction = '&document_id=' + ids.document_id;
 64 				last_transaction += '&action_id=' + ids.open_action_id;
 65 				last_transaction += '&version_id=' + ids.version_id;
 66 				last_transaction += '&document_number=' + ids.document_number;
 67 				last_transaction += '&id=' + ids.bsid;
 68 				
 69 				// Check if the last_transaction has an id
 70 				// sets last_transaction cookie if last_transaction var is populated
 71 				if(last_transaction.length !== 0){
 72 					jQuery.cookie("last_transaction",last_transaction, {path: '/'}); 
 73 				}
 74 			});
 75 		};
 76 
 77 
 78 		/**
 79 		* Callback to be passed into add_button_to_homepage. Creates an element on the page to return to quote.
 80 		* 
 81 		* @example rtq.add_button_to_homepage(rtq.v10_homepage_callback);
 82 		* @memberOf return_to_quote_button
 83 		* @param url {String} The relative url of a quote to be returned to
 84 		*/
 85 		return_to_quote_button.v10_homepage_callback = function(url,context) {
 86 			logger.debug("using v10 behavior");
 87 			context = context || document;
 88 			//we're going to attach it to the mainnav as a list item
 89 			var parent_node = jQuery("#mainnav > ul",context);
 90 
 91 			// the return to quote button is a link nested inside an li element with some styling
 92 			var child_node = "<li id='rtq_li'> <a id='rtq_link'href='"+url+"'>";
 93 			child_node += "<span class='hpinner-strip'><span class='hpinner-text'>";
 94 			child_node += "Return to Quote</span></span></a> </li>";
 95 
 96 			parent_node.append(child_node);
 97 		};
 98 	});
 99 
100 
101 	return return_to_quote_button;
102 });
