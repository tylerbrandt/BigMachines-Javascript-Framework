1 /** 
  2 * @namespace Manages common BigMachines modules
  3 * @name manager
  4 * @requires jquery
  5 * @requires logger
  6 * @requires qunit
  7 * @constructor
  8 */
  9 define(["logger","text!qunit.css","qunit"],function(logger,css) {
 10 	// define as global to create a singleton
 11 	// manager = manager || {};
 12 	var manager = {};
 13 
 14 	var modules = [];
 15 	
 16 	//can this be remote?
 17 	var module_test_url = "";
 18 
 19 	/**
 20 	* Add module to the list of modules managed by this manager and run tests if necessary.
 21 	* @memberOf manager
 22 	* @param module_name {String}: name of module to register
 23 	* @param test_module_name {String}: Optional. Name of test suite module. If not specified, defaults to {module_name}_tests
 24 	*/
 25 	manager.register = function(module_name,test_module_name) {
 26 		modules.push(module_name);
 27 
 28 		if(test_enabled()) {
 29 			// run tests
 30 			logger.debug("running tests for: " + module_name);
 31 			setup_qunit();
 32 			//require each test separately and asynchronously
 33 			var mod_test_name = test_module_name || module_test_url + module_name + "_tests";
 34 			require([mod_test_name], function(test) {
 35 				test.run_tests("#qunit-fixture");
 36 			});
 37 			
 38 		}
 39 		logger.debug("registered " + module_name + ". All Modules: " + modules);
 40 	};
 41 	
 42 	/** 
 43 	* Adds the qunit test results/CSS to the page
 44 	*/
 45 	var setup_qunit = function() {
 46 		// only one container on the page, crawl the DOM once
 47 		if (jQuery("#qunit_container").length === 0) {
 48 			logger.debug("setting up manager");
 49 			var qunitContainer = document.createElement("div");
 50 			qunitContainer.id = "qunit_container";
 51 			jQuery(qunitContainer).append(
 52 									"<h1 id='qunit-header'>Javascript Testing</h1>" + 
 53 									"<h2 id='qunit-banner'></h2>" +
 54 									"<h2 id='qunit-userAgent'></h2>" +
 55 									"<ol id='qunit-tests'></ol>" +
 56 									"<div id='qunit-fixture'>test markup, will be hidden</div>"
 57 									);
 58 			jQuery("body").append(qunitContainer);
 59 			
 60 			// css
 61 			jQuery("head").append("<style>" + css + "</style>");
 62 		}
 63 		return;
 64 	}
 65 	
 66 	/**
 67 	* Test to see if testing funtionality is on.
 68 	* @returns {Boolean} If true, run tests on modules.
 69 	*/
 70 	var test_enabled = function() {
 71 		// run tests only if the user is superuser
 72 		var user_set = typeof _BM_USER_LOGIN ==="undefined" || _BM_USER_LOGIN === "superuser";
 73 		
 74 		if(user_set) {
 75 		
 76 			// flag set in header/footer: 
 77 			var flag_set = jQuery("#mod_mgr_run_tests").attr("value") === "true";
 78 			if(flag_set) {
 79 				return true;
 80 			}
 81 		}
 82 		
 83 		return false;
 84 	};
 85 	
 86 	return manager;
 87 });
