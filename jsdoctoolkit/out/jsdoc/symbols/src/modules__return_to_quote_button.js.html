1  /**
  2   * @namespace
  3   * @name return_to_quote_button
  4   * @description A module to add a "Return To Quote" button on the homepage, based on a recent quote.
  5   * @requires jquery
  6   * @requires commerce_ids
  7   * @requires jquery.cookie
  8   **/
  9 define(["jquery", "commerce_ids", "jquery.cookie"], function($, c_ids) {
 10   var return_to_quote_button = {};
 11 
 12   /**
 13    * Called from commerce. Sets a cookie used to Return to Quote from homepage.
 14    * @memberOf return_to_quote_button
 15    * @param none {Undefined} Nothing
 16    * @return none {Undefined} Nothing
 17    */
 18   return_to_quote_button.add_button_to_homepage = function() {
 19     var last_trans_cookie = $.cookie("last_transaction");
 20 		//if cookie is set add functionality to link
 21 		if( last_trans_cookie !== null && last_trans_cookie !== "" && last_trans_cookie !== "-1" ) {
 22 			var url = '/commerce/buyside/document.jsp?formaction=performAction&';
 23 			url += last_trans_cookie;
 24 		
 25 			$(".return-to-quote").attr("href", url);
 26 			$(".return-to-quote").show();
 27 			
 28 			//parts search doesn't exist (this class is set condtionally in xsl)
 29 			if($(".return-to-quote-wrapper").length){
 30 				$(".return-to-quote-wrapper").show();
 31 			}
 32 			else{ //parts search exists
 33 				$(".return-to-quote-pipe").show();
 34 			}
 35 		}
 36 
 37   }
 38 
 39   /**
 40    * Called from homepage. Shows a return to quote button on the page, contingent
 41    * on a cookie created by set_cookie_in_commerce.
 42    * @memberOf return_to_quote_button
 43    * @param none {Undefined} Nothing
 44    * @return none {Undefined} Nothing
 45    */
 46   return_to_quote_button.set_cookie_in_commerce = function() {
 47     //get ids off page
 48     var ids = c_id.get_ids();
 49 		//builds url
 50 		var last_transaction = '&document_id=' + ids.document_id;
 51 		last_transaction += '&action_id=' + ids.open_action_id;
 52 		last_transaction += '&version_id=' + ids.version_id;
 53 		last_transaction += '&document_number=' + ids.document_number;
 54 		last_transaction += '&id=' + ids.bsid;
 55 		
 56 		// Check if the last_transaction has an id
 57 		// sets last_transaction cookie if last_transaction var is populated
 58 		if(last_transaction.length !== 0){
 59 			$.cookie("last_transaction",last_transaction, {path: '/'}); 
 60 		}
 61   };
 62 
 63 
 64   return return_to_quote_button;
 65 });
